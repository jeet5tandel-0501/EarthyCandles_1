  <script>
    let basket = JSON.parse(localStorage.getItem('basket')) || [];

    function updateBasketDisplay() {
      document.getElementById('basket-count').innerText = basket.reduce((sum, item) => sum + item.qty, 0);
      const basketItems = document.getElementById('basket-items');
      basketItems.innerHTML = '';
      if (basket.length === 0) {
        basketItems.innerHTML = '<p>Your basket is empty.</p>';
        return;
      }
      basket.forEach((item, index) => {
        const div = document.createElement('div');
        div.classList.add('basket-item');
        div.innerHTML = `
          <span>${item.qty} √ó ${item.product}</span>
          <button onclick="removeItem(${index})">Remove</button>
        `;
        basketItems.appendChild(div);
      });
    }

    function addToBasket(product, qty) {
      qty = parseInt(qty);
      const existing = basket.find(item => item.product === product);
      if (existing) {
        existing.qty += qty;
      } else {
        basket.push({ product, qty });
      }
      localStorage.setItem('basket', JSON.stringify(basket));
      updateBasketDisplay();
      alert(`${qty} √ó ${product} added to basket!`);
    }

    function removeItem(index) {
      basket.splice(index, 1);
      localStorage.setItem('basket', JSON.stringify(basket));
      updateBasketDisplay();
    }

    function clearBasket() {
      basket = [];
      localStorage.removeItem('basket');
      updateBasketDisplay();
    }

    function toggleBasket() {
      const modal = document.getElementById('basket-modal');
      modal.style.display = modal.style.display === 'flex' ? 'none' : 'flex';
      if (modal.style.display === 'flex') {
        modal.style.display = 'flex';
      }
    }

    function checkout() {
      if (basket.length === 0) {
        alert("Your basket is empty!");
        return;
      }
      alert("Proceeding to checkout... (placeholder)");
    }

    // Initial render
    updateBasketDisplay();
  </script>
